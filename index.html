1. VIP ScoreBoard

Shows "VIP" tag in the scoreboard for players with flag 't'.

#include <amxmodx> #define PLUGIN "VIP ScoreBoard" #define AUTHOR "Omar_Plugins" public plugin_init() { register_plugin(PLUGIN, "1.0", AUTHOR) register_message(get_user_msgid("ScoreAttrib"), "message_ScoreAttrib") } public message_ScoreAttrib(iMsgId, iMsgDest, iReceiver) { new id = get_msg_arg_int(1) if(is_user_connected(id) && (get_user_flags(id) & ADMIN_LEVEL_H)) { set_msg_arg_int(2, ARG_BYTE, (1<<2)) } }

2. Reset Score Pro

Type /rs to reset your score to 0-0.

#include <amxmodx> #include <cstrike> #include <fun> #define PLUGIN "Reset Score" #define AUTHOR "Omar_Plugins" public plugin_init() { register_plugin(PLUGIN, "1.1", AUTHOR) register_clcmd("say /rs", "cmd_reset") } public cmd_reset(id) { if(!is_user_alive(id)) return 1; set_user_frags(id, 0); cs_set_user_deaths(id, 0) message_begin(MSG_ALL, get_user_msgid("ScoreInfo")) write_byte(id); write_short(0); write_short(0); write_short(0); write_short(get_user_team(id)); message_end() client_print(id, print_chat, "[rs] Score Reset!") return 1 }

3. Double Jump

#include <amxmodx> #include <engine> #define PLUGIN "Double Jump" #define AUTHOR "Omar_Plugins" new bool:dojump[33] public client_PreThink(id) { if(!is_user_alive(id)) return 0 static n, o, f; n=get_user_button(id); o=get_user_oldbutton(id); f=get_entity_flags(id) if((n & IN_JUMP) && !(f & FL_ONGROUND) && !(o & IN_JUMP)) { if(dojump[id]) { dojump[id]=false; static Float:v[3]; entity_get_vector(id, EV_VEC_velocity, v) v[2]=265.0; entity_set_vector(id, EV_VEC_velocity, v) } } else if(f & FL_ONGROUND) dojump[id]=true return 0 }
