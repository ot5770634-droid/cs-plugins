1. Anti-Ping Kicker (High Latency)

Automatically kicks players with Ping higher than 150 to keep the server smooth.

#include <amxmodx> #define MAX_PING 150 #define CHECK_TIME 5.0 public plugin_init() { register_plugin("Anti-Ping Kicker", "1.0", "Omar_Plugins") set_task(CHECK_TIME, "check_ping", 0, _, _, "b") } public check_ping() { new players[32], num, id, ping, loss get_players(players, num, "ch") // Skip bots and HLTV for(new i=0; i<num; i++) { id = players[i] get_user_ping(id, ping, loss) if(ping > MAX_PING) { server_cmd("kick #%d ^"Your Ping is too high (%d)!^"", get_user_userid(id), ping) } } }
